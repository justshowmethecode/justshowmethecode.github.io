替换图片脚本

vnote默认放置图片到vnote使用下，但是hexo的是格式是文件名/图片.png，因此需要写脚本自动替换。拷贝该文件到vnote的根目录执行即可。

#!/bin/bash
# 替换vnote中的图片引用

for md_file in `find ./ -type f -name '*md'`
do
    #echo ${md_file}
    dir_name=`dirname ${md_file}`
    file_name=`basename ${md_file}|cut -d . -f 1`
    dir_path=`dirname ${md_file}`
    new_image_path="${dir_name}/${file_name}"
    mkdir ${new_image_path} &>/dev/null
    v_images="${dir_name}/vnote使用"
    #echo ${v_images}
    if [ -e ${v_images} ];then
        for image in `grep 'vnote使用' ${md_file}`
        do
            echo ${image}
            image_name=`basename ${image}|cut -d '(' -f 2|cut -d ')' -f 1`
            origin_image_path="${dir_path/2020/01/02/gong-ju/vnote/vnote-shi-yong/${image_name}"
            cp ${origin_image_path} ${new_image_path} &>/dev/null
            sed -i "/2020/01/02/gong-ju/vnote/vnote-shi-yong/${file_name}/g" ${md_file}
        done
    fi
done
